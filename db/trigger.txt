DELIMITER //
 CREATE TRIGGER generateProductNumber
 BEFORE INSERT ON products
 FOR EACH ROW
 BEGIN
 IF
 NEW.productNumber IS NULL THEN
 SET NEW.productNumber = CONCAT('PR-',UUID());
 END IF;
 END
